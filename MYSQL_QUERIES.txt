CREATE DATABASE IF NOT EXISTS shirts_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE shirts_db;

-- Categories store top-level merchandising buckets (Casual, Formal, etc.)
CREATE TABLE IF NOT EXISTS products_category (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    name VARCHAR(120) NOT NULL UNIQUE,
    slug VARCHAR(140) NOT NULL UNIQUE,
    description LONGTEXT NOT NULL,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Individual products that belong to a category
CREATE TABLE IF NOT EXISTS products_product (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    category_id BIGINT UNSIGNED NOT NULL,
    name VARCHAR(200) NOT NULL,
    slug VARCHAR(220) NOT NULL UNIQUE,
    description LONGTEXT NOT NULL,
    price DECIMAL(10,2) NOT NULL,
    stock INT UNSIGNED NOT NULL DEFAULT 0,
    size VARCHAR(4) NOT NULL,
    color VARCHAR(60) NOT NULL DEFAULT '',
    brand VARCHAR(100) NOT NULL DEFAULT '',
    image VARCHAR(255),
    is_featured TINYINT(1) NOT NULL DEFAULT 0,
    is_active TINYINT(1) NOT NULL DEFAULT 1,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    PRIMARY KEY (id),
    CONSTRAINT fk_product_category FOREIGN KEY (category_id)
        REFERENCES products_category (id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Optional gallery images per product
CREATE TABLE IF NOT EXISTS products_productimage (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    product_id BIGINT UNSIGNED NOT NULL,
    image VARCHAR(255) NOT NULL,
    alt_text VARCHAR(120) NOT NULL DEFAULT '',
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (id),
    CONSTRAINT fk_productimage_product FOREIGN KEY (product_id)
        REFERENCES products_product (id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Customer reviews per product/user pair
CREATE TABLE IF NOT EXISTS products_review (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    product_id BIGINT UNSIGNED NOT NULL,
    user_id INT UNSIGNED NOT NULL,
    rating SMALLINT UNSIGNED NOT NULL DEFAULT 5,
    comment LONGTEXT NOT NULL,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (id),
    CONSTRAINT uq_review_product_user UNIQUE (product_id, user_id),
    CONSTRAINT fk_review_product FOREIGN KEY (product_id)
        REFERENCES products_product (id) ON DELETE CASCADE,
    CONSTRAINT fk_review_user FOREIGN KEY (user_id)
        REFERENCES auth_user (id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Cart header, 1:1 with each authenticated user
CREATE TABLE IF NOT EXISTS cart_cart (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    user_id INT UNSIGNED NOT NULL UNIQUE,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    PRIMARY KEY (id),
    CONSTRAINT fk_cart_user FOREIGN KEY (user_id)
        REFERENCES auth_user (id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Line items stored per cart/product combination
CREATE TABLE IF NOT EXISTS cart_cartitem (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    cart_id BIGINT UNSIGNED NOT NULL,
    product_id BIGINT UNSIGNED NOT NULL,
    quantity INT UNSIGNED NOT NULL DEFAULT 1,
    added_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (id),
    CONSTRAINT uq_cartitem UNIQUE (cart_id, product_id),
    CONSTRAINT fk_cartitem_cart FOREIGN KEY (cart_id)
        REFERENCES cart_cart (id) ON DELETE CASCADE,
    CONSTRAINT fk_cartitem_product FOREIGN KEY (product_id)
        REFERENCES products_product (id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Saved addresses for checkout
CREATE TABLE IF NOT EXISTS orders_address (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    user_id INT UNSIGNED NOT NULL,
    full_name VARCHAR(150) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    address_line1 VARCHAR(255) NOT NULL,
    address_line2 VARCHAR(255) NOT NULL DEFAULT '',
    city VARCHAR(100) NOT NULL,
    state VARCHAR(100) NOT NULL,
    postal_code VARCHAR(20) NOT NULL,
    country VARCHAR(100) NOT NULL DEFAULT 'India',
    is_default TINYINT(1) NOT NULL DEFAULT 0,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    PRIMARY KEY (id),
    CONSTRAINT fk_address_user FOREIGN KEY (user_id)
        REFERENCES auth_user (id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Order header rows, linked to addresses and users
CREATE TABLE IF NOT EXISTS orders_order (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    user_id INT UNSIGNED NOT NULL,
    order_number VARCHAR(20) NOT NULL UNIQUE,
    address_id BIGINT UNSIGNED NOT NULL,
    status VARCHAR(20) NOT NULL DEFAULT 'pending',
    subtotal DECIMAL(10,2) NOT NULL,
    shipping DECIMAL(10,2) NOT NULL DEFAULT 0,
    total_amount DECIMAL(10,2) NOT NULL,
    placed_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    PRIMARY KEY (id),
    CONSTRAINT fk_order_user FOREIGN KEY (user_id)
        REFERENCES auth_user (id) ON DELETE CASCADE,
    CONSTRAINT fk_order_address FOREIGN KEY (address_id)
        REFERENCES orders_address (id) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Items captured per order, product pointer optional for historical data
CREATE TABLE IF NOT EXISTS orders_orderitem (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    order_id BIGINT UNSIGNED NOT NULL,
    product_id BIGINT UNSIGNED,
    product_name VARCHAR(200) NOT NULL,
    quantity INT UNSIGNED NOT NULL DEFAULT 1,
    price DECIMAL(10,2) NOT NULL,
    PRIMARY KEY (id),
    CONSTRAINT fk_orderitem_order FOREIGN KEY (order_id)
        REFERENCES orders_order (id) ON DELETE CASCADE,
    CONSTRAINT fk_orderitem_product FOREIGN KEY (product_id)
        REFERENCES products_product (id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Extended profile info per user
CREATE TABLE IF NOT EXISTS profiles_profile (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    user_id INT UNSIGNED NOT NULL UNIQUE,
    phone VARCHAR(20) NOT NULL DEFAULT '',
    bio LONGTEXT NOT NULL,
    avatar VARCHAR(255),
    updated_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    PRIMARY KEY (id),
    CONSTRAINT fk_profile_user FOREIGN KEY (user_id)
        REFERENCES auth_user (id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Wishlists map a user to many products
CREATE TABLE IF NOT EXISTS profiles_wishlist (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    user_id INT UNSIGNED NOT NULL UNIQUE,
    updated_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    PRIMARY KEY (id),
    CONSTRAINT fk_wishlist_user FOREIGN KEY (user_id)
        REFERENCES auth_user (id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE IF NOT EXISTS profiles_wishlist_products (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    wishlist_id BIGINT UNSIGNED NOT NULL,
    product_id BIGINT UNSIGNED NOT NULL,
    PRIMARY KEY (id),
    CONSTRAINT uq_wishlist_product UNIQUE (wishlist_id, product_id),
    CONSTRAINT fk_wishlist_products_wishlist FOREIGN KEY (wishlist_id)
        REFERENCES profiles_wishlist (id) ON DELETE CASCADE,
    CONSTRAINT fk_wishlist_products_product FOREIGN KEY (product_id)
        REFERENCES products_product (id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Browse master data
SELECT id, name, slug, created_at FROM products_category ORDER BY name;
SELECT id, name, price, stock, brand, is_featured FROM products_product ORDER BY created_at DESC LIMIT 20;

-- Inspect product reviews with author info
SELECT r.id, p.name AS product, u.username AS reviewer, r.rating, r.created_at
FROM products_review r
JOIN products_product p ON r.product_id = p.id
JOIN auth_user u ON r.user_id = u.id
ORDER BY r.created_at DESC
LIMIT 50;

-- View carts along with total items per user
SELECT u.username, c.id AS cart_id, SUM(ci.quantity) AS total_items
FROM cart_cart c
JOIN auth_user u ON c.user_id = u.id
LEFT JOIN cart_cartitem ci ON ci.cart_id = c.id
GROUP BY c.id, u.username
ORDER BY c.updated_at DESC;

-- Shipping addresses per user
SELECT u.username, a.full_name, a.city, a.state, a.is_default
FROM orders_address a
JOIN auth_user u ON a.user_id = u.id
ORDER BY a.updated_at DESC;

-- High-level order list with totals
SELECT o.order_number, u.username, o.status, o.total_amount, o.placed_at
FROM orders_order o
JOIN auth_user u ON o.user_id = u.id
ORDER BY o.placed_at DESC
LIMIT 100;

-- Items within a single order (replace placeholder with actual number)
SELECT oi.product_name, oi.quantity, oi.price, (oi.quantity * oi.price) AS line_total
FROM orders_orderitem oi
JOIN orders_order o ON oi.order_id = o.id
WHERE o.order_number = 'REPLACE_WITH_ORDER_NO';

-- Wishlist contents per user
SELECT u.username, p.name AS product_name
FROM profiles_wishlist w
JOIN auth_user u ON w.user_id = u.id
JOIN profiles_wishlist_products wp ON wp.wishlist_id = w.id
JOIN products_product p ON wp.product_id = p.id
ORDER BY u.username, p.name;
